    \vspace{-5mm} \section*{0}
    次の不定積分を求めて下さい。(積分定数として$C, C_1, C_2, \cdots $を断らずに用いてよい)

    \begin{description}
        \setlength{\itemsep}{0.5cm}

        \begin{multicols}{2}

            \item[(1)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int x^2\cos{x} \ dx \\
                    & \hspace*{2mm} = x^2\sin{x} - 2 \int x\sin{x} \ dx \\
                    & \hspace*{2mm} = x^2\sin{x} - 2\sin{x} + 2x\cos{x} \\
                    & \hspace*{2mm} = (x^2 - 2)\sin{x} + 2x\cos{x} + C
                \end{flalign*}


            \item[(2)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int\left(log{x}\right)^2 \ dx \\
                    & \hspace*{2mm} = xlog^2{x} - 2\int log{x} \ dx \\
                    & \hspace*{2mm} = xlog^2{x} - 2\left( xlog{x} - x \right) \\
                    & \hspace*{2mm} = x\left(log^2{x} - 2log{x} + 2 \right) + C_1
                \end{flalign*}

        \end{multicols}

        \begin{multicols}{2}

            \item[(3)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int\cfrac{x}{\sqrt{1-x^2}} \ dx \cdots (*) \\
                    & \hspace*{2mm} u = 1 - x^2 とすると \\
                    & \hspace*{2mm} (*) = -\cfrac{1}{2} \int\cfrac{1}{\sqrt{u}} \ du \\
                    & \hspace*{6.5mm} = -\sqrt{u} + C_2 \\
                    & \hspace*{6.5mm} = -\sqrt{1 - x^2} + C_2
                \end{flalign*}

            \item[(4)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int\cfrac{\sin^{-1}{x}}{\sqrt{1-x^2}} \ dx \cdots (*) \\
                    & \hspace*{2mm} u = \sin^{-1}{x}とすると \\
                    & \hspace*{2mm} (*) = \int u \ du \\
                    & \hspace*{6.5mm} = \cfrac{u^2}{2} + C_3 \\
                    & \hspace*{6.5mm} = \cfrac{\sin^{-1}{x^2}}{2} + C_3
                \end{flalign*}

        \end{multicols}

        \item[(5)]
            \begin{flalign*}
                & \hspace*{-10mm} \int \tan^{-1}{x} \ dx \\
                & \hspace*{2mm} = x\tan^{-1}{x} - \int \cfrac{x}{x^2 + 1} \ dx \cdots (*) \\
                & \hspace*{2mm} u = x^2 + 1  とすると \\
                & \hspace*{2mm} (*) = x\tan^{-1}{x} - \cfrac{1}{2} \int \cfrac{1}{u} \ du \\
                & \hspace*{6.5mm} = x\tan^{-1}{x} - \cfrac{log{u}}{2} + C_4 \\
                & \hspace*{6.5mm} = x\tan^{-1}{x} - \cfrac{log{(x^2  + 1)}}{2} + C_4
            \end{flalign*}

    \end{description}

    \section*{1}
    次の不定積分を求めて下さい。(積分定数として$C, C_1, C_2, \cdots $を断らずに用いてよい)

    \begin{description}
        \setlength{\itemsep}{0.5cm}

        \begin{multicols}{2}

            \item[(1)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int \cfrac{x^3 + 2x^2 + 4x + 6}{x^2 + 4} \ dx \\
                    & \hspace*{-2mm} = \int \left(-\cfrac{2}{x^2 + 4} + x + 2 \right) \ dx \\
                    & \hspace*{-2mm} = -2\int \cfrac{1}{x^2 + 4} \ dx + \int x \ dx + 2\int 1 \ dx \\
                    & \hspace*{-2mm} = -\tan^{-1}{\cfrac{x}{2}} + \cfrac{x^2}{2} + 2x + C \\
                    & \hspace*{-2mm} = \cfrac{x}{2}(x + 4) - \tan^{-1}{\cfrac{x}{2}} + C
                \end{flalign*}

            \item[(2)]
                \begin{flalign*}
                    & \hspace*{-10mm} \int \cfrac{x^2 + 2x + 1}{x^2 - 2x + 2} \ dx \\
                    & \hspace*{-2mm} = \int \left(\cfrac{4x - 1}{x^2 - 2x + 2} + 1\right) \ dx \\
                    & \hspace*{-2mm} = \int \cfrac{4x - 1}{x^2 - 2x + 2} \ dx + \int 1 \ dx \\
                    & \hspace*{-2mm} = \int \left(\cfrac{2(2x-2)}{x^2-2x+2} + \cfrac{3}{x^2-2x+2} \right) \ dx + x + C_1 \\
                    & \hspace*{-2mm} = 2\int \cfrac{2x-2}{x^2-2x+2} \ dx + 3\int \cfrac{1}{x^2-2x+2} \ dx + x + C_1 \cdots (*) \\
                    & \hspace*{-2mm} u = x^2 - 2x + 2 とすると \\
                    & \hspace*{-2mm} (*) = 2log{u} + 3\int \cfrac{1}{(x-1)^2 + 1} \ dx + x + C_1 \cdots (**) \\
                    & \hspace*{-2mm} t = x - 1 とすると \\
                    & \hspace*{-2mm} (**) = 2log{(x^2 - 2x + 2)} + 3\int \cfrac{1}{t^2 + 1}\ dt + x + C_1 \\
                    & \hspace*{4.5mm} = 2log{(x^2 - 2x + 2)} - 3\tan^{-1}{t} + x + C_1 \\
                    & \hspace*{4.5mm} = 2log{(x^2 - 2x + 2)} - 3\tan^{-1}{(1-x)} + x + C_1
                \end{flalign*}

        \end{multicols}

        \item[(3)]
            \begin{flalign*}
                & \hspace*{-10mm} \int\cfrac{2}{(x-1)(x+1)^2} \ dx \\
                & \hspace*{-2mm} = 2\int \left(-\cfrac{1}{4(x+1)} - \cfrac{1}{2(x+1)^2} + \cfrac{1}{4(x-1)}\right) \ dx \\
                & \hspace*{-2mm} = -\cfrac{1}{2}\int\cfrac{1}{x+1}\ dx - \int\cfrac{1}{(x+1)^2}\ dx + \cfrac{1}{2}\int\cfrac{1}{x-1}\ dx \cdots (*) \\
                & \hspace*{-2mm} u = x+1, t = x-1 とすると \\
                & \hspace*{-2mm} (*) = -\cfrac{1}{2}\int\cfrac{1}{u}\ du - \int\cfrac{1}{u^2}\ du + \cfrac{1}{2}\int\cfrac{1}{t}\ dt \\
                & \hspace*{3mm} = -\cfrac{log{u}}{2} + \cfrac{1}{u} + \cfrac{log{t}}{2} + C_2 \\
                & \hspace*{3mm} = -\cfrac{log{(x+1)}}{2} + \cfrac{1}{x+1} + \cfrac{log{(x-1)}}{2} + C_2 \\
                & \hspace*{3mm} = \cfrac{1}{2}\left(\cfrac{2}{x+1} + log{(x-1)} - log{(x+1)} \right) + C_2
            \end{flalign*}

        \item[(4)]
            \begin{flalign*}
                & \hspace*{-10mm} \int\cfrac{1}{2x^2-x} \ dx \\
                & \hspace*{-2mm} = \int\cfrac{1}{\left(\sqrt{2}x - \cfrac{1}{2\sqrt{2}}\right)^2 - \cfrac{1}{8}} \ dx \cdots (*) \\
                & \hspace*{-2mm} u = \sqrt{2}x - \cfrac{1}{2\sqrt{2}} とすると \\
                & \hspace*{-2mm} (*) = \cfrac{1}{\sqrt{2}} \int \cfrac{1}{u^2 - \cfrac{1}{8}}\ du \\
                & \hspace*{3mm} = \cfrac{1}{\sqrt{2}} \int \cfrac{8}{8u^2 - 1}\ du \\
                & \hspace*{3mm} = 4\sqrt{2} \int \cfrac{1}{8u^2 - 1}\ du \\
                & \hspace*{3mm} = -4\sqrt{2} \int \cfrac{1}{1 - 8u^2}\ du \cdots (**) \\
                & \hspace*{-2mm} t = 2\sqrt{2}u とすると \\
                & \hspace*{-2mm} (**) = -2\int\cfrac{1}{1-t^2}\ dt \\
                & \hspace*{4.5mm} = -2\tanh^{-1}{t} + C_3 \\
                & \hspace*{4.5mm} = log{(1-t)} - log{(1+t)} + C_3 \\
                & \hspace*{4.5mm} = log{\left(1-2\sqrt{2}u\right)} - log{\left(1+2\sqrt{2}u\right)} + C_3 \\
                & \hspace*{4.5mm} = log{\left(1-2\sqrt{2}\left(\sqrt{2}x - \cfrac{1}{2\sqrt{2}}\right)\right)} - log{\left(1+2\sqrt{2}\left(\sqrt{2}x - \cfrac{1}{2\sqrt{2}}\right)\right)} + C_3 \\
                & \hspace*{4.5mm} = log{(2-4x)} - log{4x} + C_3 \\
                & \hspace*{4.5mm} = log{(1-2x)} - log{2x} + C_3
            \end{flalign*}

        \item[(5)]
            \begin{flalign*}
                & \hspace*{-10mm} \int\cfrac{4}{(x-1)(x^2+1)}\ dx \\
                & \hspace*{-2mm} = 4\int\left(\cfrac{-x-1}{2(x^2+1)} + \cfrac{1}{2(x-1)}\right)\ dx \\
                & \hspace*{-2mm} = 2\int\cfrac{-x-1}{x^2+1}\ dx + 2\int\cfrac{1}{x-1}\ dx \\
                & \hspace*{-2mm} = 2\int\left(-\cfrac{x}{x^2+1}-\cfrac{1}{x^2+1}\right)\ dx + 2log{(x-1)} + C_4 \\
                & \hspace*{-2mm} = -2\int\cfrac{x}{x^2+1}\ dx - 2\int\cfrac{1}{x^2+1}\ dx + 2log{(x-1)} + C_4 \cdots (*) \\
                & \hspace*{-2mm} u = x^2 + 1 とすると \\
                & \hspace*{-2mm} (*) = -\int\cfrac{1}{u}\ du - 2\tan^{-1}{x} + 2log{(x-1)} + C_4 \\
                & \hspace*{3mm} = -log{u} - 2\tan^{-1}{x} + 2log{(x-1)} + C_4 \\
                & \hspace*{3mm} = -log{(x^2+1)} - 2\tan^{-1}{x} + 2log{(x-1)} + C_4
            \end{flalign*}

    \end{description}
